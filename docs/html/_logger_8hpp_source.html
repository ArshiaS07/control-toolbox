100 const LogEntry& get_by_time(double t) const {
101 auto it = std::lower_bound(entries_.begin(), entries_.end(), t,
102 [](const LogEntry& e, double t) { return e.timestamp < t; });
103
104 if (it != entries_.end() && it->timestamp == t) {
105 return *it;
106 }
107 throw std::out_of_range("No log entry at requested time");
108 }

160 std::vector<double> get_state_component(size_t idx, double t_start, double t_end) const {
161 std::vector<double> values;
162 for (const auto& entry : entries_) {
163 if (entry.timestamp >= t_start && entry.timestamp <= t_end) {
164 values.push_back(entry.state[idx]);

177 double calculate_amplitude(size_t state_idx, double t_start, double t_end) const {
178 auto values = get_state_component(state_idx, t_start, t_end);
179 if (values.empty()) return 0.0;
180
181 auto [min_it, max_it] = std::minmax_element(values.begin(), values.end());
182 return *max_it - *min_it;
183 }

193 void write_header(std::ostream& os) const {
194 os << "time";

196 // State components

213 // Output components

216 os << ",ref" << i;

241 for (int i = 0; i < Traits::InputDimension; ++i)
242 os << "," << entry.control_input[i];

245 os << "," << entry.disturbed_input[i];
246 }
247
248 // Outputs
249 if constexpr (Traits::OutputDimension > 0) {
250 for (int i = 0; i < Traits::OutputDimension; ++i)
251 os << "," << entry.reference[i];
252
253 for (int i = 0; i < Traits::OutputDimension; ++i)
254 os << "," << entry.output_clean[i];
255
256 for (int i = 0; i < Traits::OutputDimension; ++i)
257 os << "," << entry.output_noisy[i];
