161 StateVector computeSystemDerivative(double t, const StateVector& x, bool log = true) {
162 SignalFlowData flow;
163 flow.timestamp = t;
164 flow.state = x;
165
166 // 1. Reference Signal Generation (Tracking Systems)
167 if constexpr (Traits::HasOutput) {
168 flow.reference = static_cast<Derived*>(this)->computeReference(t);
169 }
170
171 // 2. Previous Output Handling (One-Step Delay)
172 OutputVector y_clean = OutputVector::Zero();
173 if constexpr (Traits::HasOutput) {
174 if (previous_output_.size() > 0) {
175 y_clean = previous_output_;
176 }
177 flow.output_clean = y_clean;
178 }
179
180 // 3. Measurement Noise Injection
181 OutputVector y_noisy = y_clean;
182 if constexpr (Traits::HasOutput) {
183 y_noisy = static_cast<Derived*>(this)->addOutputNoise(y_clean, t);
184 flow.output_noisy = y_noisy;
185 }
186
187 // 4. Control Signal Computation
188 InputVector u = InputVector::Zero();
189 if constexpr (Traits::HasInput) {
190 if constexpr (Traits::HasOutput) {
191 // Tracking Control: u = f(ref, y, x)
192 u = static_cast<Derived*>(this)->computeController(t, flow.reference, y_noisy, x);
193 }
194 else {
195 // Regulation Control: u = f(x)
196 u = static_cast<Derived*>(this)->computeController(t, x);
197 }
198 flow.control_input = u;
199 }
200
201 // 5. Input Disturbance Injection
202 InputVector u_disturbed = u;
203 if constexpr (Traits::HasInput) {
204 u_disturbed = static_cast<Derived*>(this)->addInputDisturbance(u, t);
205 flow.disturbed_input = u_disturbed;
206 }
207
208 // 6. System Dynamics Computation
209 flow.derivative = static_cast<Derived*>(this)->computeDynamics(t, x, u_disturbed);
210
211 // 7. Current Output Update (For Next Step)
212 if constexpr (Traits::HasOutput) {
213 previous_output_ = static_cast<Derived*>(this)->computeOutput(t, x, u);
214 }
215
216 // Data Logging
217 if (log) {
218 struct LoggerType::LogEntry entry;
219 entry.timestamp = flow.timestamp;
220 entry.state = flow.state;
221 entry.derivative = flow.derivative;
222 entry.control_input = flow.control_input;
223 entry.disturbed_input = flow.disturbed_input;
224 entry.reference = flow.reference;
225 entry.output_clean = flow.output_clean;
226 entry.output_noisy = flow.output_noisy;
227 logger_.log(entry);
228 }
229
230 return flow.derivative;
231 }
